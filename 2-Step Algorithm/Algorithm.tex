\section{Overview}
% =================================================================

The schedule and the base camps are coupled: KPIs measuring travel,
jet-lag, geographic dispersion, altitude, and entry/visa exposure depend
jointly on which slot/stadium each match is assigned to \emph{and} where
each team is based. We solve the coupled problem by
\textbf{alternating minimisation} with \textbf{stochastic exploration}:

\begin{description}[leftmargin=1.4em, itemsep=3pt]
  \item[\textbf{Step A (Schedule step).}] Hold base camps fixed; solve a
        mixed-integer linear programme (MILP) that assigns matches to
        slots and stadiums to minimise a \emph{weighted sum} of the
        thirteen selected KPIs, subject to the hard constraints H1--H8.
  \item[\textbf{Step B (Base-camp step).}] Hold the schedule fixed;
        re-optimise each team's base camp under travel cost and the US
        entry/visa rules, with a Metropolis exploration move.
\end{description}

The thirteen KPIs entering the objective are
1.2, 1.3, 1.4, 1.6, 1.7, 2.2, 2.4, 3.3, 4.1, 4.2, 5.1, 5.2, 5.3.

\section{Decision Variables}
% =================================================================

\paragraph{Upper level (schedule).}
\begin{align}
x_{mts} &=
  \begin{cases} 1 & m \text{ in slot } t,\text{ stadium } s\\ 0 &
  \text{else}\end{cases}
  &&\forall m\in M,\,t\in T,\,s\in S,\\
y_{gt} &=
  \begin{cases} 1 & t \text{ is the final slot of group } g\\ 0 &
  \text{else}\end{cases}
  &&\forall g\in G,\,t\in T.
\end{align}

\paragraph{Lower level (base camp).}
\begin{align}
u_{ib} &=
  \begin{cases} 1 & \text{team } i \text{ is based at facility }
    b\in B_i\\ 0 & \text{else}\end{cases}
  &&\forall i\in I,\,b\in B_i.
\end{align}
In Step~A the base camp is fixed, $u=\hat u$, so $b_i$ and hence
$\dist(b_i,\cdot),\tzone(b_i),\elev(b_i)$ are constants; in Step~B the
schedule is fixed and $u$ is the decision.

\section{Hard Constraints}
% =================================================================
We restate the feasibility model in the present notation. Here $H\subset
I$ is the set of host-nation teams and $S_i\subseteq S$ the stadiums of
team $i$'s home country.

\paragraph{H1 -- Each match scheduled once.}
\begin{equation}
\sum_{t\in T}\sum_{s\in S} x_{mts}=1 \qquad\forall m\in M.
\end{equation}
\paragraph{H2 -- Round-robin incidence.}
\begin{equation}
\sum_{m\in M_i}\sum_{t\in T}\sum_{s\in S} x_{mts}=3 \qquad\forall i\in I.
\end{equation}
\paragraph{H3 -- One match per team per round.}
\begin{equation}
\sum_{m\in M_i}\sum_{t\in T_r}\sum_{s\in S} x_{mts}=1
  \qquad\forall i\in I,\;r\in\{1,2,3\}.
\end{equation}
\paragraph{H4 -- Stadium turnover.} A stadium hosts at most one match in
any rolling window of length $\Delta_{\min}+105/60$:
\begin{equation}
\sum_{m\in M}\;\sum_{\substack{t'\in T\\
  \tau_t\le\tau_{t'}<\tau_t+\Delta_{\min}+105/60}}
  x_{mt's}\le 1 \qquad\forall s\in S,\,t\in T.
\end{equation}
\paragraph{H5 -- Minimum team rest (72 h kickoff-to-kickoff $+$ duration).}
\begin{equation}
\sum_{s} x_{mts}+\sum_{s} x_{m't's}\le 1
  \quad
  \begin{aligned}[t]
  &\forall i\in I,\;\forall m,m'\in M_i,\,m\neq m',\\
  &\forall t,t'\in T:\,|\tau_t-\tau_{t'}|<R_{\min}+105/60.
  \end{aligned}
\end{equation}
\paragraph{H6 -- Host-nation venue restriction.}
\begin{equation}
\sum_{t\in T}\sum_{s\in S\setminus S_i} x_{mts}=0
  \qquad\forall i\in H,\,m\in M_i.
\end{equation}
\paragraph{H7 -- Simultaneous final matches.}
\begin{align}
\sum_{t\in T} y_{gt}&=1 &&\forall g\in G,\\
\sum_{m\in M_g}\sum_{s} x_{mts}&\ge 2\,y_{gt} &&\forall g\in G,\,t\in T,\\
\sum_{m\in M_g}\sum_{s}\sum_{\substack{t'\in T\\ \tau_{t'}>\tau_t}} x_{mt's}
  &\le 2\,(1-y_{gt}) &&\forall g\in G,\,t\in T.
\end{align}
\paragraph{H8 -- Match allocation by host country.}
\begin{equation}
\sum_{m\in M}\sum_{t\in T}\sum_{s\in S_c} x_{mts}=N_c
  \qquad\forall c\in\{\mathrm{USA,MEX,CAN}\}.
\end{equation}

% =================================================================
\section{KPI Terms in the Schedule Objective}
\label{sec:kpi}
% =================================================================

Each KPI is written as a function of the schedule variables $x$
(with base camps fixed at $\hat b$ in Step~A) and then normalised by its
current-schedule value $\overline{K}_\bullet$ so that the weighted sum is
dimensionless. KPIs are oriented so that \emph{smaller is better};
maximise-type KPIs (5.1, 5.2, 5.3) enter as ``shortfall'' terms. Each
$\dist(\hat b_i,s)$, $\elev(s)$, $W_{mt}$, $q_t$ multiplying a binary
$x_{mts}$ is a known constant, so every term is linear after the
standard auxiliary-variable linearisations.

\paragraph{KPI 1.2 -- Intra-group travel dispersion.}
Per-team base-camp round-trip travel
$\mathrm{TD}_i=2\sum_{m\in M_i}\sum_{t,s}\dist(\hat b_i,s)\,x_{mts}.$
Group \emph{range} surrogate for the coefficient of variation: with
$\mathrm{TD}_g^{\max}\ge \mathrm{TD}_i\ge \mathrm{TD}_g^{\min}\ \forall
i\in g$,
\begin{equation}
  K_{1.2}=\sum_{g\in G}\bigl(\mathrm{TD}_g^{\max}-\mathrm{TD}_g^{\min}\bigr).
\end{equation}

\paragraph{KPI 1.3 -- Circadian shift cost.}
With base-camp time zone fixed, the perceived-night penalty
$\phi_{its}=\phi\!\bigl((\theta_t-[\tzone(s)-\tzone(\hat b_i)])\bmod 24\bigr)$
is a constant per (team, slot, stadium). Hence
\begin{equation}
  K_{1.3}=\sum_{i\in I}\sum_{m\in M_i}\sum_{t,s}\phi_{its}\,x_{mts}.
\end{equation}

\paragraph{KPI 1.4 -- Match-venue geographic dispersion.}
Let $a_{ic}\in\{0,1\}$ indicate team $i$ visits cluster $c$:
$a_{ic}\ge x_{mts}\ \forall m\in M_i,\,s:\cl(s)=c$. Then
\begin{equation}
  K_{1.4}=\sum_{i\in I}\sum_{c\in\{\mathrm W,\mathrm C,\mathrm E\}} a_{ic}.
\end{equation}

\paragraph{KPI 1.6 -- Rest asymmetry between opponents.}
Let $r_{mi}$ be team $i$'s rest hours before match $m$ (linear in $x$
via slot times). With $\Delta_m\ge r_{m,i(m)}-r_{m,j(m)}$ and
$\Delta_m\ge r_{m,j(m)}-r_{m,i(m)}$,
\begin{equation}
  K_{1.6}=\sum_{m\in M}\Delta_m .
\end{equation}

\paragraph{KPI 1.7 -- Entry and visa restriction exposure.}
Severity-weighted count of affected teams playing at US stadiums:
\begin{equation}
  K_{1.7}=\sum_{i\in I}\sigma_i
          \sum_{m\in M_i}\sum_{t}\sum_{s\in S^{\mathrm{US}}} x_{mts}.
\end{equation}

\paragraph{KPI 2.2 -- Per-team heat load.}
With excess WBGT $h_{mt}=\max(0,W_{mt}-28)$ a constant,
\begin{equation}
  K_{2.2}=\sum_{i\in I}\sum_{m\in M_i}\sum_{t,s} h_{mt}\,x_{mts}.
\end{equation}

\paragraph{KPI 2.4 -- Altitude disruption.}
With base-camp elevation fixed, $f_{is}=f(|\elev(s)-\elev(\hat b_i)|)$
(the KPI 2.4 penalisation function) is constant, so
\begin{equation}
  K_{2.4}=\sum_{i\in I}\sum_{m\in M_i}\sum_{t,s} f_{is}\,x_{mts}.
\end{equation}

\paragraph{KPI 3.3 -- Round-order balance (first-mover).}
Let $a_i=\sum_{r\in\{1,2\}}\mathrm{pos}(i,r)$ count earlier-kickoff
appearances in the two sequential rounds ($\mathrm{pos}(i,r)$ linear in
$x$ and the round-slot ordering). Balancing targets $a_i=1$; with
$e_i\ge a_i-1$, $e_i\ge 1-a_i$,
\begin{equation}
  K_{3.3}=\sum_{i\in I} e_i .
\end{equation}

\paragraph{KPI 4.1 -- Venue-load balance.}
Let $n_s=\sum_{m,t} x_{mts}$ and mean $\bar n=|M|/|S|$. Mean-absolute-
deviation surrogate for the CV with $d_s\ge n_s-\bar n$,
$d_s\ge \bar n-n_s$:
\begin{equation}
  K_{4.1}=\sum_{s\in S} d_s .
\end{equation}

\paragraph{KPI 4.2 -- Same-city overlap.}
With $o_{ct}\ge \sum_{m}\sum_{s:\,\mathrm{city}(s)=c} x_{mts}-1$,
$o_{ct}\ge 0$ counting simultaneous same-city matches,
\begin{equation}
  K_{4.2}=\sum_{c}\sum_{t} o_{ct}.
\end{equation}

\paragraph{KPI 5.1 -- Prime-time alignment (maximise).}
Enter as shortfall below the current prime-time value
$\overline{\mathrm{PT}}$:
\begin{equation}
  K_{5.1}=\overline{\mathrm{PT}}-\sum_{m\in M}\sum_{t,s} q_t\,x_{mts}.
\end{equation}

\paragraph{KPI 5.2 -- Marquee-match slot quality (maximise) with overlap.}
\begin{equation}
  K_{5.2}=\overline{\mathrm{MSQ}}
          -\sum_{m\in M}\sum_{t,s}\mu_m\,q_t\,x_{mts}
          +\lambda\!\!\sum_{\{m,m'\}\subseteq P}\!\! o^P_{mm'},
\end{equation}
where $o^P_{mm'}\ge \sum_s x_{mts}+\sum_s x_{m'ts}-1$ at each common slot
$t$ penalises two high-profile matches sharing a slot.

\paragraph{KPI 5.3 -- Host-city economic equity (maximise).}
Venue commercial value $\mathrm{VC}_s=\sum_{m,t}\mu_m q_t x_{mts}$,
mean $\overline{\mathrm{VC}}=\tfrac1{|S|}\sum_s\mathrm{VC}_s$.
Gini-replacing mean-absolute-deviation dispersion with
$p_s\ge \mathrm{VC}_s-\overline{\mathrm{VC}}$,
$p_s\ge \overline{\mathrm{VC}}-\mathrm{VC}_s$:
\begin{equation}
  K_{5.3}=\sum_{s\in S} p_s .
\end{equation}

\section{Step A --- Schedule MILP (Weighted-Sum Objective)}
\label{sec:stepA}
% =================================================================

With base camps fixed at $\hat b$, all KPI coefficients are constants
and every $K_\bullet$ is linear. The schedule subproblem is:
\begin{align}
  \min_{x,y,\text{aux}}\quad
  Z(x;\hat b)\;=\;
  &\;w_{1.2}\tfrac{K_{1.2}}{\overline{K}_{1.2}}
   +w_{1.3}\tfrac{K_{1.3}}{\overline{K}_{1.3}}
   +w_{1.4}\tfrac{K_{1.4}}{\overline{K}_{1.4}}
   +w_{1.6}\tfrac{K_{1.6}}{\overline{K}_{1.6}}
   +w_{1.7}\tfrac{K_{1.7}}{\overline{K}_{1.7}}
   \notag\\
  &+w_{2.2}\tfrac{K_{2.2}}{\overline{K}_{2.2}}
   +w_{2.4}\tfrac{K_{2.4}}{\overline{K}_{2.4}}
   +w_{3.3}\tfrac{K_{3.3}}{\overline{K}_{3.3}}
   +w_{4.1}\tfrac{K_{4.1}}{\overline{K}_{4.1}}
   +w_{4.2}\tfrac{K_{4.2}}{\overline{K}_{4.2}}
   \notag\\
  &+w_{5.1}\tfrac{K_{5.1}}{\overline{K}_{5.1}}
   +w_{5.2}\tfrac{K_{5.2}}{\overline{K}_{5.2}}
   +w_{5.3}\tfrac{K_{5.3}}{\overline{K}_{5.3}}
  \label{eq:Zobj}\\
  \text{s.t.}\quad & \text{H1--H8 and all auxiliary definitions of
    Section~\ref{sec:kpi},}\notag
\end{align}
with $w_\bullet\ge 0$, $\sum_\bullet w_\bullet=1$. Denote the optimal
schedule $x^\star(\hat b)$ with value $Z^\star(\hat b)$.

\section{Step B --- Base-Camp Step with US Ban / Bond Logic}
\label{sec:stepB}
% =================================================================

Given the fixed schedule, every match's stadium and slot are known, so
each team's base-camp cost decomposes \emph{by team}. For team $i$ and
candidate facility $b\in B_i$, the cost combines travel and the US
entry/visa rule:
\begin{equation}
  C_i(b)\;=\;
    \underbrace{2\!\!\sum_{m\in M_i}\!\dist(b,s(m))}_{\text{round-trip travel}}
    \;+\;\Pi_i(b),
  \label{eq:Cib}
\end{equation}
where $s(m)$ is the fixed stadium of match $m$ and the \textbf{US
base-camp rule} is
\begin{equation}
  \Pi_i(b)\;=\;
  \begin{cases}
    +\infty & i\in F^{\mathrm{ban}}\ \text{and}\ b\in\Omega^{\mathrm{US}}
              \quad\text{(banned teams cannot camp in the US)}\\[4pt]
    \pi^{\mathrm{bond}} & i\in F^{\mathrm{bond}}\ \text{and}\
              b\in\Omega^{\mathrm{US}}
              \quad\text{(bond teams penalised for a US camp)}\\[4pt]
    0 & \text{otherwise.}
  \end{cases}
  \label{eq:visa}
\end{equation}
Equivalently in assignment form: fully banned teams have US facilities
removed from the menu, $u_{ib}=0\ \forall b\in\Omega^{\mathrm{US}},\,i\in
F^{\mathrm{ban}}$; bond teams incur the additive $\pi^{\mathrm{bond}}$
km-equivalent whenever $b\in\Omega^{\mathrm{US}}$.

\paragraph{Greedy (exploitation) move.}
$b_i^{\mathrm{greedy}}=\arg\min_{b\in B_i} C_i(b)$, an $O(|B_i|)$
enumeration. The $+\infty$ branch guarantees a banned team never selects
a US camp; the $\pi^{\mathrm{bond}}$ branch makes a US camp selectable
for a bond team only when its travel saving exceeds the penalty.

\paragraph{Stochastic exploration move.}
With annealed temperature $\theta_t>0$:
\begin{enumerate}[leftmargin=*, itemsep=2pt]
  \item w.p.\ $1-p_{\mathrm{exp}}$ propose $b'=b_i^{\mathrm{greedy}}$;
        w.p.\ $p_{\mathrm{exp}}$ propose $b'\sim\mathrm{Uniform}(B_i)$.
        For $i\in F^{\mathrm{ban}}$, draw only from
        $B_i\setminus\Omega^{\mathrm{US}}$ so banned-team proposals stay
        feasible.
  \item $\Delta_i=C_i(b')-C_i(b_i^{\mathrm{cur}})$; accept
        $b_i\leftarrow b'$ w.p.\ $\min\{1,\exp(-\Delta_i/\theta_t)\}$
        (Metropolis), else keep.
\end{enumerate}
Because $\pi^{\mathrm{bond}}$ is on the same km scale as travel, a bond
team trades a US camp's geographic convenience against the
fan-accessibility cost it imposes (KPI~1.7).

\section{Iterative Algorithm}
\label{sec:algorithm}
% =================================================================

\begin{algorithm}[h]
\DontPrintSemicolon
\caption{Two-Step Iterative Base-Camp Optimisation with Exploration}
\KwIn{Current base camps $b^{(0)}$; menus $\{B_i\}$; KPI weights
      $\{w_\bullet\}$; bond penalty $\pi^{\mathrm{bond}}$; sets
      $F^{\mathrm{ban}},F^{\mathrm{bond}}$;
      $\theta_0,\rho,\theta_{\min},p_{\mathrm{exp}}$; caps
      $T_{\max},P$.}
\KwOut{Best schedule $x^\star$ and base camps $b^\star$.}
\BlankLine
Compute normalisers $\overline{K}_\bullet,\overline{\mathrm{PT}},
  \overline{\mathrm{MSQ}},\overline{\mathrm{VC}}$ from the current
  schedule.\;
$b\leftarrow b^{(0)}$;\quad
$x\leftarrow \textsc{ScheduleMILP}(b)$\tcp*{Step A, eq.~\eqref{eq:Zobj}}
$Z^\star\leftarrow Z(x;b)$;\quad
$(b^\star,x^\star)\leftarrow(b,x)$;\quad
$\theta\leftarrow\theta_0$;\quad$\mathrm{stall}\leftarrow0$\;
\For{$k\leftarrow1$ \KwTo $T_{\max}$}{
  \ForEach{$i\in I$}{
    $b_i^{\mathrm{greedy}}\leftarrow\arg\min_{b\in B_i} C_i(b)$
       \tcp*{eq.~\eqref{eq:Cib}--\eqref{eq:visa}}
    \eIf{$\mathrm{rand}()<p_{\mathrm{exp}}$}{
      $b'\leftarrow\textsc{UniformRandom}\bigl(\,
        i\in F^{\mathrm{ban}}\,?\,B_i\setminus\Omega^{\mathrm{US}}
        :B_i\,\bigr)$\tcp*{explore (feasible)}
    }{
      $b'\leftarrow b_i^{\mathrm{greedy}}$\tcp*{exploit}
    }
    $\Delta_i\leftarrow C_i(b')-C_i(b_i)$\;
    \lIf{$\mathrm{rand}()<\min\{1,\exp(-\Delta_i/\theta)\}$}{
      $b_i\leftarrow b'$}
  }
  $x\leftarrow\textsc{ScheduleMILP}(b)$\tcp*{Step A again}
  \eIf{$Z(x;b)<Z^\star$}{
    $Z^\star\leftarrow Z(x;b)$;\;
    $(b^\star,x^\star)\leftarrow(b,x)$;\;
    $\mathrm{stall}\leftarrow0$\;
  }{
    $\mathrm{stall}\leftarrow\mathrm{stall}+1$\;
  }
  $\theta\leftarrow\rho\,\theta$\;
  \lIf{$\mathrm{stall}\ge P$ \textbf{and} $\theta<\theta_{\min}$}{
    \textbf{break}}
}
\Return $(b^\star,x^\star)$\;
\end{algorithm}

\noindent
\textsc{ScheduleMILP}$(b)$ solves \eqref{eq:Zobj} subject to H1--H8. The
incumbent updates only on strict improvement, so $Z^\star$ is
non-increasing and the returned solution is feasible and never worse
than the current-schedule baseline. With $p_{\mathrm{exp}}=0,
\theta\to0$ the method reduces to deterministic alternating minimisation
converging to a coordinate-wise local optimum; the annealed exploration
provides the global-search mechanism over $\prod_i B_i$.
